  while(true){
    if(p.HP==0)l.text="GAMEOVER rを押すとリトライ";

while(true){
  if(getkey("right")>0)	x+=3;
  if(getkey("left")>0)	x-=3;
  if(getkey("down")>0)	y+=3;
  if(getkey("up")>0)	y-=3;
  if(crashTo(Enemy))    HP-=1;
  if(HP==0)             die();
  update();
}
