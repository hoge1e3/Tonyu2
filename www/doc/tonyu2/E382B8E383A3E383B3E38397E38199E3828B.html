用途別リファレンス

ジャンプする

  • contactToで、他の物理オブジェクトと触れているかどうかを判定できます。

Ball
extends BodyActor;
shape="circle";
while(true) {
    if (getkey("left")) vx=-3;   
    if (getkey("right")) vx=3;   
    // スペースキーが押されていて、しかもFloorと触れているときに上に力を加える
    if (getkey("space")==1 && contactTo(Floor)) vy=-10;
    update();
}
